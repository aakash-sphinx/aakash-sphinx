## Install

```bash
npm install --save sawin-plugin
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'sawin-plugin'
import 'sawin-plugin/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
